local Library = loadstring(game:HttpGet("https://raw.githubusercontent.com/bloodball/-back-ups-for-libs/main/wizard"))()
local Window = Library:NewWindow("Wizard Clicker Simulator")

local Tab1 = Window:NewSection("Main")

Tab1:CreateToggle("Auto Cheapest Egg", function(bool)
        AutoHatchCheapestEgg = bool
        while AutoHatchCheapestEgg do
            game:GetService("ReplicatedStorage").Packages._Index:FindFirstChild("sleitnick_knit@1.4.7").knit.Services.EggService.RE.Purchase:FireServer("Rainbow",1)
            task.wait()
        end
end)

Tab1:CreateToggle("Auto Defeat Bosses", function(bool)
        AutoDefeatBoss = bool
        while AutoDefeatBoss do
            fireproximityprompt(workspace.__WORLDS.Grassland.Interactive.Mats["1"].Interactive.Prompt.ProximityPrompt)            
            workspace.__WORLDS.Grassland.Interactive.Mats["1"].Interactive.Prompt.ProximityPrompt.HoldDuration = 0
            game:GetService("ReplicatedStorage").Packages._Index:FindFirstChild("sleitnick_knit@1.4.7").knit.Services.BattleService.RE.Click:FireServer()
            task.wait()
        end
end)

Tab1:CreateButton("Anti-AFK", function()
game:GetService("StarterGui"):SetCore("SendNotification", { Title = "Execute Successful", Text = "Anti-AFK Enabled", Duration = 15 })
if getconnections then
	for i,v in next, getconnections(game.Players.LocalPlayer.Idled) do
		v:Disable()
	end
end
end)
